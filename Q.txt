int longestSubarrayWithSumK(vector<int> a, long long k) {
    // Write your code here
    int maxlength = 0;
    for(int i = 0;i<a.size();i++){
        int currentSum =0;
        for(int j =i;j<a.size();j++){
            currentSum +=a[j];
            if(currentSum == k){
             	 maxlength = max(maxlength,j-i+1);
            }
        }
    }
    return maxlength;
}
